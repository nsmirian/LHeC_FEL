### Fiel function #####
def Und_Field( c1, c2,c3,g, lambda0):
    B=c1 *np.exp(c2*(g/lambda0)+c3*(g/lambda0)**2)
    return B
c1=1.45; c2=-1.28; c3=-2.24;  lambda0=1.6e-2  # Helical 
cp1=1.96; cp2=-0.82; cp3=-3.31; 
g = np.arange(0.0028, 0.01, 0.0001)
#----------------------------
gamma=80000
Kp=0.934*Und_Field( cp1, cp2,cp3,g, lambda0)*lambda0*100
Kh=0.934*Und_Field( c1, c2,c3,g, lambda0)*lambda0*100
LamP=lambda0/(2*gamma**2) *(1+Kp**2/2)*1e10
LamH=lambda0/(2*gamma**2) *(1+Kh**2)*1e10
#--------------------------
plt.figure(1)
ax1 = plt.subplot(211)
ax1.set_xlabel('gap[mm]')
ax1.set_ylabel('B[T]')
ax1.plot(g*2, Und_Field( c1, c2,c3,g, lambda0), 'b',\
         g*2, Und_Field( cp1, cp2,cp3,g, lambda0), '--b')
ax1.tick_params(axis='y', labelcolor='k')
ax2 = ax1.twinx() 
ax2.set_ylabel('K')  
ax2.plot( g*2, 0.934*Kh,'g',label='Helical')
ax2.plot( g*2, 0.934*Kp, '--g', label='Planar' )
plt.legend();# fig.tight_layout()
plt.grid(b=True,axis='both',  color='#666666', linestyle=':')
###
ax1 = plt.subplot(212)
ax1.set_xlabel('gap[mm]')
ax1.set_ylabel('$\lambda[A]$', color='b')
ax1.plot(g*2, LamH, 'b',\
         g*2, LamP, '--b')
ax1.tick_params(axis='y', labelcolor='b')
ax2 = ax1.twinx() 
ax2.set_ylabel('K', color='g')  
ax2.plot( g*2, Kh,'g',label='Helical')
ax2.plot( g*2,Kp, '--g', label='Planar' )
ax2.tick_params(axis='y', labelcolor='g')
plt.legend()
#fig.tight_layout()
plt.grid(b=True,axis='both',  color='#666666', linestyle=':')
plt.show()
